Code Lauren introspection prototype

ball: make-circle(50 50 20)
set(ball "x-speed" 0)
set(ball "y-speed" 0)

gravity: 0.5

forever {
  bat: make-rectangle(mouse-x mouse-y 100 10)

  if are-overlapping(bat ball) {
    bat-ball-difference: subtract(get(ball "x") get(bat "x"))
    set(ball "x-speed" divide(bat-ball-difference 10))
    set(ball "y-speed" opposite(get(ball "y-speed")))
  } else {
    set(ball "y-speed" add(gravity get(ball "y-speed")))
  }

  set(ball "x" add(get(ball "x") get(ball "x-speed")))
  set(ball "y" add(get(ball "y") get(ball "y-speed")))

  clear-screen()
  draw(ball)
  draw(bat)
}
